## Objetivo

bandit22@bandit:~$ ls /etc/cron.d
cronjob_bandit15_root  cronjob_bandit22  cronjob_bandit24       e2scrub_all  sysstat

echo "Copying passwordfile /etc/bandit_pass/$myname to /tmp/$mytarget"
